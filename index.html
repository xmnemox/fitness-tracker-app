        // Funzione per convertire il peso da kg a libbre

        function kgToLbs(kg) {

            return kg * 2.20462;

        }


        // Funzione per convertire la lunghezza da cm a pollici

        function cmToInches(cm) {

            return cm * 0.393701;

        }


        // Funzione per calcolare il BMI/IMC

        function calculateBmi(weightKg, heightCm) {

            if (weightKg <= 0 || heightCm <= 0) return null;

            const heightM = heightCm / 100;

            return (weightKg / (heightM * heightM)).toFixed(2);

        }


        // Funzione per calcolare la massa in kg da una percentuale

        function calculateMass(weightKg, percentage) {

            if (weightKg <= 0 || percentage < 0) return null;

            return (weightKg * (percentage / 100)).toFixed(2);

        }

        // Ascoltatore per la cronologia completa dei dati

        async function setupHistoryListener() {

            const q = query(collection(db, 'artifacts', appId, 'users', userId, 'daily_data'), orderBy('data', 'desc'));

            onSnapshot(q, (querySnapshot) => {

                const history = [];

                const weights = [];

                const ingestedCalories = [];

                const consumedCalories = [];

                const dates = [];

                querySnapshot.forEach((doc) => {

                    const data = doc.data();

                    history.push(data);

                    dates.push(data.data);

                    weights.push(data.peso);

                    ingestedCalories.push(data.kcalIngerite);

                    consumedCalories.push(data.kcalConsumate);

                });


                renderHistory(history);

                updateCharts(dates, weights, ingestedCalories, consumedCalories);

                updateWeeklySummary(history);

            });

        }


        // Funzione per renderizzare la cronologia

        function renderHistory(history) {

            const historyContainer = document.getElementById('historyContainer');

            historyContainer.innerHTML = ''; // Pulisci la cronologia esistente

            history.forEach(data => {

                const entry = document.createElement('div');

                entry.className = 'p-4 border border-gray-200 rounded-lg shadow-sm bg-white mb-2';

                entry.innerHTML = `

                    <h3 class="font-semibold text-lg text-gray-800">${data.data}</h3>

                    <p><strong>Peso:</strong> ${data.peso} kg</p>

                    <p><strong>BMI/IMC:</strong> ${calculateBmi(data.peso, userGoals.altezza)}</p>

                    <p><strong>Kcal Ingerite:</strong> ${data.kcalIngerite} kcal</p>

                    <p><strong>Kcal Consumate:</strong> ${data.kcalConsumate} kcal</p>

                    <p><strong>Acqua Assunta:</strong> ${data.acquaAssunta} L</p>

                    <button class="mt-2 text-blue-500 hover:text-blue-700" onclick="showDailyData('${data.data}')">Modifica</button>

                `;

                historyContainer.appendChild(entry);

            });

        }

        // Funzione per calcolare e mostrare il riassunto settimanale

        function updateWeeklySummary(history) {

            // Logica semplificata: calcola la media della settimana più recente

            const weeklyData = history.slice(0, 7);

            if (weeklyData.length === 0) return;


            const totalWeight = weeklyData.reduce((sum, entry) => sum + (entry.peso || 0), 0);

            const totalKcalIngested = weeklyData.reduce((sum, entry) => sum + (entry.kcalIngerite || 0), 0);

            const totalKcalConsumed = weeklyData.reduce((sum, entry) => sum + (entry.kcalConsumate || 0), 0);


            const avgWeight = (totalWeight / weeklyData.length).toFixed(2);

            const avgKcalIngested = (totalKcalIngested / weeklyData.length).toFixed(2);

            const avgKcalConsumed = (totalKcalConsumed / weeklyData.length).toFixed(2);

            

            document.getElementById('summary-dates').textContent = `Dati dal ${weeklyData[weeklyData.length - 1].data} al ${weeklyData[0].data}`;

            document.getElementById('summary-weight').textContent = avgWeight;

            document.getElementById('summary-kcal-ingested').textContent = avgKcalIngested;

            document.getElementById('summary-kcal-consumed').textContent = avgKcalConsumed;

        }


        // Funzione per cambiare scheda

        function showTab(tabId) {

            const tabs = ['daily', 'weekly', 'graphs', 'history'];

            tabs.forEach(tab => {

                document.getElementById(tab).classList.add('hidden');

                document.getElementById(`${tab}-tab`).classList.remove('bg-blue-600', 'text-white');

                document.getElementById(`${tab}-tab`).classList.add('bg-gray-200', 'text-gray-700');

            });


            document.getElementById(tabId).classList.remove('hidden');

            document.getElementById(`${tabId}-tab`).classList.remove('bg-gray-200', 'text-gray-700');

            document.getElementById(`${tabId}-tab`).classList.add('bg-blue-600', 'text-white');

            

            if (tabId === 'daily') {

                 // Aggiorna la data all'oggi se la scheda daily è selezionata

                document.getElementById('data').valueAsDate = new Date();

            }

        }
